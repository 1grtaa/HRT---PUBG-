const ccxt = require('ccxt'); let io; const initSocket = (server) => { io = require('socket.io')(server, { cors: { origin: '*' } }); io.on('connection', (socket) => { console.log('🔌 New WebSocket connection'); let intervals = []; socket.on('subscribe', async (symbols) => { const symbolList = symbols.split(','); const binance = new ccxt.binance(); // إرسال الأسعار كل 3 ثواني const interval = setInterval(async () => { try { for (const symbol of symbolList) { const ticker = await binance.fetchTicker(symbol.trim()); socket.emit('ticker', { symbol: ticker.symbol, last: ticker.last, change: ticker.percentage, high: ticker.high, low: ticker.low }); } } catch (e) { console.error('Ticker error:', e.message); } }, 3000); intervals.push(interval); }); socket.on('disconnect', () => { intervals.forEach(clearInterval); console.log('🔌 Client disconnected'); }); }); }; const emitToUser = (userId, event, data) => { if (io) { io.emit(`user:${userId}:${event}`, data); } }; const broadcastSignal = (signal) => { if (io) { io.emit('new_signal', signal); } }; module.exports = { initSocket, emitToUser, broadcastSignal };
